mean squared error

def mse(x: ComplexFloat, y: ComplexFloat) -> float:
    """mean squared error"""
    return ((x - y) ** 2).mean()
sax.nn.loss.huber_loss(x: Union[complex, float, jax._src.numpy.lax_numpy.ndarray, numpy.ndarray], y: Union[complex, float, jax._src.numpy.lax_numpy.ndarray, numpy.ndarray], delta: float = 0.5) → float#

huber loss

def huber_loss(x: ComplexFloat, y: ComplexFloat, delta: float=0.5) -> float:
    """huber loss"""
    return ((delta ** 2) * ((1.0 + ((x - y) / delta) ** 2) ** 0.5 - 1.0)).mean()

The huber loss is like the mean squared error close to zero and mean absolute error for outliers

sax.nn.loss.l2_reg(weights: Dict[str, Union[complex, float, jax._src.numpy.lax_numpy.ndarray, numpy.ndarray]]) → float#

L2 regularization loss

def l2_reg(weights: Dict[str, ComplexFloat]) -> float:
    """L2 regularization loss"""
    numel = 0
    loss = 0.0
    for w in (v for k, v in weights.items() if k[0] in ("w", "b")):
        numel = numel + w.size
        loss = loss + (jnp.abs(w) ** 2).sum()
    return loss / numel
